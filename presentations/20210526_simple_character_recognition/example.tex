\iffalse
this example shows how to increase tracking (space between letters) and turn off ligature
\fi
\documentclass{letter}
\usepackage{lmodern}
\usepackage[T1]{fontenc}
\usepackage{xcolor}
\usepackage[tracking=true, letterspace = 100]{microtype}
\DisableLigatures{encoding = *, family = *}
\begin{document}
\normalfont
\color{blue}
\textls{Through three cheese trees\ \\
three free fleas flew\ \\
While these fleas flew\ \\
freezy breeze blew\ \\
Freezy breeze made\ \\
these three trees freeze\ \\
Freezy trees made\ \\
these trees cheese freeze\ \\
Thats what made these\ \\
three free fleas sneeze\ \\
}
\end{document}
